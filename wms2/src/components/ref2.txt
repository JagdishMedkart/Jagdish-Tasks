import React from "react";
import styles from "../../styles/ProductListing.module.scss";
import { productStructure } from "@/utils/productStructure"; // adjust the path as needed

const ProductTable = ({ products }) => {
  return (
    <table className={styles.table}>
      <thead>
        <tr>
          {productStructure.headers.map((header) => (
            <th key={header.key} className={styles.th}>
              {header.display}
            </th>
          ))}
        </tr>
      </thead>
      <tbody>
        {products.map((product, rowIndex) => (
          <tr key={`product-${rowIndex}`}>
            {productStructure.headers.map((header) => {
              const value = product[header.key];
              let extraClass = "";
              if (header.key === "publish_status" && header.values && header.values[value]) {
                // header.values[value] is a string like "styles.publishedDot"
                // We remove the "styles." prefix and use the remaining string to access the class from styles.
                const classNameKey = header.values[value].replace("styles.", "");
                extraClass = styles[classNameKey] || "";
              }
              return (
                <td key={`${header.key}-${product.product_id}`} className={`${styles.td} ${extraClass}`}>
                  {value}
                </td>
              );
            })}
          </tr>
        ))}
      </tbody>
    </table>
  );
};

export default ProductTable;
